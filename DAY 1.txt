#include<iostream>
using namespace std;

void printarray(int arr[] , int size)
{
    for(int i=0;i<size;i++)
    {
        cout<<arr[i];
    }
}
void swapalternate(int arr[] , int size)
{
    for(int i=0;i<size;i+=2)
    {
        if(i+1<size)
        {
            swap(arr[i],arr[i+1]);
        }
    }
}
int main()
{
    int arr[5]={1,3,5,7,9};
    swapalternate(arr,5);
    printarray(arr,5);
}
